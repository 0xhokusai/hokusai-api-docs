Set Royalty

POST /v1/nfts/{contractId}/{tokenId}/royalty#

Set NFT royalty.

Request Parameter#

ParameterTypeExample ValueDescriptionRequired
contractIdstringADXS6zyen4NctAjHcontractId issued by Hokusai APIRequired
tokenIdstring1tokenId issued by Hokusai API when minting NFTRequired

Request Body#

KeyTypeExample ValueDescription
receiverstring0x3a93da588954acf4d0d8f1f1a4439fa79d84cf29Royalty recipient's Ethereum address
percentagenumber10Royalty rate (0~100%)

Response#

KeyTypeExample ValueDescription
txHashstring0xc968510337593d167efa433ab3726634cab8754d88f38e6e61e8ac7071601c42Hash value of the transaction that issues the NFT

curl -X POST -H "Content-Type: application/json" \ "https://api.hokusai.app/v1/nfts/ADXS6zyen4NctAjH/1/royalty?key=fKe2rDFUZjSSdXfW" \-d '{"receiver":"0x3a93da588954acf4d0d8f1f1a4439fa79d84cf29", "percentage": 10}'
const baseUrl = "https://api.hokusai.app"
const mintNft = async (baseUrl, apiKey, contractId, tokenId, receiver, percentage) => {  const path = `/v1/nfts/${contractId}/${tokenId}/royalty`  const url = new URL(baseUrl + path);
  const params = {    key: apiKey  };
  const requestBody = {    receiver,    percentage  };
  url.search = new URLSearchParams(params).toString();
  const res = await fetch(url, {    method: 'POST',    headers: {      'Content-Type': 'application/json'    },    body: JSON.stringify(requestBody)  });  return res.json();}
import requests
base_url = "https://api.hokusai.app"
def set_royalty(base_url: str, api_key: str, contract_id: str, token_id: str, receiver: str, percentage: float):  path = f"/v1/nfts/{contract_id}/{token_id}/royalty"
  params = {    "key": api_key  }
  request_body = {    "receiver": receiver,    "percentage": percentage,  }
  res = requests.post(base_url + path, params=params, json=request_body)  return res.json()
